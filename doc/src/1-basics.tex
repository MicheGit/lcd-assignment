        \begin{flalign*}
            P ::= & \overline{x}\,v.P \\
             & x(x).P \\
             & P \pipe P \\
             & \mathtt{if}\,v\,\mathtt{then}\,P\,\mathtt{else}\,P \\
             & 0 \\
             & (\nu\,x\,x)P \\
             \\
            v ::= & x \\
            & \mathtt{true}\\
            & \mathtt{false}
        \end{flalign*}

    \begin{minipage}{0.45\textwidth}

        \begin{lstlisting}{language=pilang}
            P ::= x << v . P
                | x >> x . P
        \end{lstlisting}
